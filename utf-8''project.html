The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

# Set up environment and define User Defined Functions
import zipfile
from PIL import Image
import pytesseract as pt
import cv2 as cv
import numpy as np
import math
import os

# User defined function that takes in a file containing images and parameters required for detection and finds faces in the 
# image file. It returns an array that contains the location of faces in the file
face_cascade = cv.CascadeClassifier('readonly/haarcascade_frontalface_default.xml')
def detectFaces(image_file, scale_factor, min_neighbors, min_size):
    
    # read in the file as an OpenCV object
    cv_image = cv.imread(image_file)
    gray = cv.cvtColor(cv_image, cv.COLOR_BGR2GRAY)    
    faces = face_cascade.detectMultiScale(gray,
                                          scaleFactor = scale_factor,
                                          minNeighbors = min_neighbors,
                                          minSize = min_size)
    
    return faces


# User defined function that takes in an image with identified faces and a list of face locations and makes a contact sheet.
# The alloted space for each image is 100x100 pixels (though it may be smaller) and there are 5 images per row.
def contactSheet(face_file, face_array):
    
    # Figure out how many rows it needs to have, assuming the images are square (i.e. max size is 100x100 pixels)
    cols = 5
    rows = math.ceil(len(face_array)/5)
    contact_sheet = Image.new('RGB', (cols*100, rows*100))

    # Loop through the face array and resize and paste into the contact sheet. Take advantage of the fact that the face definitions 
    # are square by default. 
    paste_x = 0
    paste_y = 0
    
    for (x, y, w, h) in face_array:
        crop_box = (x, y, x + w, y + h)
        image_to_crop = Image.open(face_file)
        face = image_to_crop.crop(crop_box)
        width = face.size[0]
        
        # if the width is greater than 100 pixels (also means hieght is since they are square, just resize)
        if width > 100:
            face_thumbnail = face.copy()
            face_thumbnail.thumbnail((100, 100))
            contact_sheet.paste(face_thumbnail, (paste_x, paste_y)) 
    
        # otherwise, just paste in the image as it is
        else:
            contact_sheet.paste(face, (paste_x, paste_y))
    
        # update the coordinates for pasting... remember that the image always gets 100x100 pixels but can be smaller
        if paste_x < 400:
            paste_x += 100
            paste_y = paste_y
        else:
            paste_x = 0
            paste_y += 100
    
    # Return an image object that contains the contact sheet
    return contact_sheet
 

# User defined function that filters faces from a contact sheet to reduce the number of false positives
def filterFaces(contact_sheet):
    
    # Take in a contact sheet file
    cv_cs = cv.imread(contact_sheet)
    faces_2 = detectFaces(contact_sheet, 2.0, 3, (30, 30))
    
    # if there are faces detected in the contact sheet image, then create a new, more accurate contact sheet
    if len(faces_2) > 0:
        
        # if there's already a file that's a place holder of this function, delete it
        if os.path.exists("cs_detected.jpg"):
            os.remove("cs_detected.jpg")
        
        # createa file that's a place holder of this function that is used to create an image object (contact sheet) out
        # of a OpenCV object
        cv.imwrite('cs_detected.jpg', cv_cs)
        filtered_line_up = contactSheet('cs_detected.jpg', faces_2)
        
        return filtered_line_up
            
    else:
        return None
    
    
# User defined function that takes in a dictionary of images and a string and returns faces on the page if that string exists

         # if the file test.jpg exists, delete it, because that's going to hold a newspaper image loop executes
